            base.OnViewLoaded(view);
            _mainViewModel = IoC.Get<MainViewModel>("mainViewModel");
            ActivateItem(new LoginViewModel());
        }


        public void Handle(Navigation message)
        {
            if (message.Page == NagivationPages.Main)
            {
                ActivateItem(_mainViewModel);
            }
            if (message.Page == NagivationPages.Details)
            {
                ActivateItem(new AppointmentViewModel(int.Parse(message.Args[0]),int.Parse(message.Args[1])));
            }
        }

        public void Handle(string message)
        {
            Username = message;
        }

        public void Logout()
        {
            IoC.Get<UserData>("UserData").DoctorId = 0;
            IoC.Get<UserData>("UserData").UserName = "";
            IoC.Get<UserData>("UserData").Password = "";
            ActivateItem(new LoginViewModel());
        }
    }
}


namespace LekarzCore.ViewModels
{
    public class MainViewModel: Screen
    {
        private readonly IEventAggregator _ev;

        private CancellationTokenSource _cancellationTokenSource;

       ...

        public MainViewModel()
        {
            _ev = IoC.Get<IEventAggregator>("eventAggregator");            
        }

        public void ViewDetails(Appointment sender)
        {
            _ev.PublishOnUIThread(new Navigation(NagivationPages.Details, new []{ sender.Id.ToString(), sender.PatientId.ToString() }));
        }

        protected override void OnActivate()
        {
            Appointments=new BindableCollection<Appointment>();
            _cancellationTokenSource = new CancellationTokenSource();
            PeriodicUpdateDataAsync(TimeSpan.FromSeconds(30), _cancellationTokenSource.Token);
            base.OnActivate();
            
        }

        protected override void OnDeactivate(bool close)
        {
            _cancellationTokenSource.Cancel();
            base.OnDeactivate(close);
        }


        public async Task PeriodicUpdateDataAsync(TimeSpan interval, CancellationToken token)
        {
            while (!token.IsCancellationRequested)
            {
                Loading = true;
                var client = new LekarzServiceClient();
                client.ClientCredentials.UserName.UserName = IoC.Get<UserData>("UserData").UserName;
                client.ClientCredentials.UserName.Password = IoC.Get<UserData>("UserData").Password;
                var result = await client.GetAppointmentsAsync(IoC.Get<UserData>("UserData").ClinicId, IoC.Get<UserData>("UserData").DoctorId);

                Appointments = new BindableCollection<Appointment>(result);

                client.Close();
                Loading = false;

                if (interval > TimeSpan.Zero)
                {
                    await Task.Delay(interval, token);
                }
            }
           
        }
    }
}

Aplikacja stworzona na potrzeby portalu Project Sin do tworzenia losowych drużyn. Chętni zgłaszali swoje uczestwictwo poprzez formularz Google Forms. Program odczytuje dane za pośrednictwem Google Api i tworzy n-osobowe drużyny według ustawień korzystając z losowego ciągu pobieranego ze strony Random.org. public static class Simple
{
    public static async Task<IEnumerable<object>> ShuffleListAsync(IEnumerable<object> listToShuffle, TimeSpan timeout)
    {
        if (!listToShuffle.Any()) return new List<object>();


        var toShuffle = listToShuffle as IList<object> ?? listToShuffle.ToList();

        var getDelayedLocalVal = new Task<Task<IEnumerable<int>>>(async() =>
        {
            await Task.Delay(timeout);

            var random = new Random.Org.Random { UseLocalMode = true };

            return random.RandomizeSequence(0, toShuffle.Count() - 1);

        });

        var getRandomFromWeb = new Task<Task<IEnumerable<int>>>(async () =>
        {
            var random = new Random.Org.Random { UseLocalMode = false };
            try
            {
                return random.RandomizeSequence(0, toShuffle.Count() - 1);
            }
            catch (WebException exception)
            {
                Trace.WriteLine(exception);
                await Task.Delay(TimeSpan.FromMilliseconds(timeout.TotalMilliseconds * 2));
                return await getDelayedLocalVal.Result;
            }

        });

        getDelayedLocalVal.Start();
        getRandomFromWeb.Start();

        IEnumerable<int> randomnumbers = await Task.WhenAny(getRandomFromWeb.Result, getDelayedLocalVal.Result).Result;

        return randomnumbers.Select(randomnumber => toShuffle[randomnumber]).ToList();
    }
}


public static class Complex
{
    private static IEnumerable<GroupSetting> _groups;
    private static IEnumerable<SummonerData> _summonerData;
    public async static Task<IEnumerable<SummonerData>> Shuffle(IEnumerable<SummonerData> Summoners, IEnumerable<GroupSetting> Groups, TimeSpan timeout)
    {
        _summonerData = Summoners;
        _groups = Groups;
        Summoners = (await Simple.ShuffleListAsync(Summoners, timeout))
                                 .OfType<SummonerData>()
                                 .ToList();
        return CreateTeams(DivideSummoner(Summoners));
    }
    private static IEnumerable<SummonerData> CreateTeams(List<List<SummonerData>> SummonerGroups)
    {
        var resultlist = new List<SummonerData>();
        while (resultlist.Count < _summonerData.Count())
        {
            var iteration = 0;
            foreach (var group in _groups)
            {
                for (var i = 0; i < group.Number; i++)
                {
                    if (SummonerGroups[iteration].Any())
                    {
                        resultlist.Add(SummonerGroups[iteration].First());
                        SummonerGroups[iteration].Remove(SummonerGroups[iteration].First());
                    }
                }
                iteration++;
            }
        }
        return resultlist;
    }
    private static List<List<SummonerData>> DivideSummoner(IEnumerable<SummonerData> Summoners)
    {
        var resultList = new List<List<SummonerData>>();
        var i = 0;
        foreach (var group in _groups)
        {
            resultList.Add(new List<SummonerData>());
            foreach (var summoner in Summoners)
            {
                if (summoner.Division >= group.From && summoner.Division <= group.To)
                    resultList[i].Add(summoner);
            }
            i++;
        }
        return resultList;
    }
